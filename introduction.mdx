  <Steps>
    <Step title="Install the package">
      Install effect-orpc along with its peer dependencies.

      <CodeGroup>
      ```bash npm
      npm install effect-orpc effect @orpc/server @orpc/client @orpc/contract @orpc/shared
      ```
      ```bash yarn
      yarn add effect-orpc effect @orpc/server @orpc/client @orpc/contract @orpc/shared
      ```
      ```bash pnpm
      pnpm add effect-orpc effect @orpc/server @orpc/client @orpc/contract @orpc/shared
      ```
      ```bash bun
      bun add effect-orpc effect @orpc/server @orpc/client @orpc/contract @orpc/shared
      ```
      </CodeGroup>
    </Step>

    <Step title="Create your runtime and services">
      Define your Effect services and create a ManagedRuntime.

      ```typescript
      import { Effect, ManagedRuntime } from "effect";

      class UserService extends Effect.Service<UserService>()("UserService", {
        accessors: true,
        sync: () => ({
          get: (id: string) => users.find((u) => u.id === id),
        }),
      }) {}

      const runtime = ManagedRuntime.make(UserService.Default);
      ```
    </Step>

    <Step title="Create Effect-aware procedures">
      Use `makeEffectORPC` to create procedures with full Effect integration.

      ```typescript
      import { makeEffectORPC, ORPCTaggedError } from "effect-orpc";
      import { z } from "zod";

      class UserNotFoundError extends ORPCTaggedError("UserNotFoundError", {
        status: 404,
      }) {}

      const effectOs = makeEffectORPC(runtime).errors({ UserNotFoundError });

      const getUser = effectOs
        .input(z.object({ id: z.string() }))
        .effect(function* ({ input }) {
          const user = yield* UserService.get(input.id);
          if (!user) {
            return yield* new UserNotFoundError();
          }
          return user;
        });
      ```
    </Step>

    <Step title="Build your router">
      Export your router and you're ready to serve it.

      ```typescript
      export const router = {
        users: {
          get: getUser,
        },
      };

      export type Router = typeof router;
      ```
    </Step>
  </Steps>
